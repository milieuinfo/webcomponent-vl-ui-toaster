              const newWarningAlert = () => { return document.querySelector('#alert-1').cloneNode(true); }
              const newErrorAlert = () => { return document.querySelector('#alert-2').cloneNode(true); }
              const newSuccessAlert = () => { return document.querySelector('#alert-3').cloneNode(true); }
              const newCtaAlert = () => { return document.querySelector('#alert-4').cloneNode(true); }
            </script>

Combineren met alerts


vl-toaster component aanvaard enkel vl-alert component.

      
        <script>
          const demoToaster = () => { return document.querySelector("#demo-toaster") };
        </script>

        <div id="demo-toaster" is="vl-toaster"></div>
          <button is="vl-button" onclick="demoToaster().push(newSuccessAlert());">Success</button>
          <button is="vl-button" onclick="demoToaster().push(newWarningAlert());">Warning</button>
          <button is="vl-button" onclick="demoToaster().push(newErrorAlert());">Error</button>
          <button is="vl-button" onclick="demoToaster().push(newCtaAlert());">Call of action</button>
        </div>

Positioneren

        
          <script>
            const topLeftToaster = () =>{ return document.querySelector("#top-left-toaster")};
            const topRightToaster = () =>{ return document.querySelector("#top-right-toaster")};
            const bottomLeftToaster = () =>{ return document.querySelector("#bottom-left-toaster")};
            const bottomRightToaster = () =>{ return document.querySelector("#bottom-right-toaster")};
          </script>

          <div id="top-left-toaster" is="vl-toaster" top-left></div>
          <div id="top-right-toaster" is="vl-toaster" top-right></div>
          <div id="bottom-left-toaster" is="vl-toaster" bottom-left></div>
          <div id="bottom-right-toaster" is="vl-toaster" bottom-right></div>
          <button onclick="topLeftToaster().push(newSuccessAlert());">Top-left</button>
          <button onclick="topRightToaster().push(newSuccessAlert());">Top-right</button>
          <button onclick="bottomLeftToaster().push(newSuccessAlert());">Bottom-left</button>
          <button onclick="bottomRightToaster().push(newSuccessAlert());">Bottom-right</button>
