vl-ui-toaster demo

Fadeout

      
        <script>
          const createErrorAlert = () => {
            return document
                    .createRange()
                      .createContextualFragment(`<vl-alert type="error" icon="warning" title="Error!" closable>
                                                  <p>Er is een fout opgetreden.</p>
                                                 </vl-alert>`)
          };

          const onError = (e) => {
            document.querySelector("#top-left-toaster").addAlert(createErrorAlert());
          };

          const createSuccessAlert = () => {
            return document
              .createRange()
                .createContextualFragment(`<vl-alert type="success" icon="check" title="Gelukt!" closable>
                                          <p>Wij hebben uw melding goed ontvangen en nemen deze spoedig in behandeling.</p>
                                         </vl-alert>`)
          };

          const onSuccess = (e) => {
            document.querySelector("#top-left-toaster").addAlert(createSuccessAlert());
          }
        </script>
        <div id="top-left-toaster" is="vl-toaster" top-left fadeout></div>
        <button onclick="onError()">Error!</button>
        <button onclick="onSuccess()">Success!</button>
