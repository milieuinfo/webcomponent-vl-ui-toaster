vl-ui-toaster demo

Fadeout

      
      <script>
        const topLeftToaster = () =>{ return document.querySelector("#top-left-toaster")};
  
        const newErrorAlert = () => {
          return document
            .createRange()
              .createContextualFragment(`<vl-alert type="error" icon="warning" title="Error!" closable>
                                          <p>Er is een fout opgetreden.</p>
                                         </vl-alert>`)
        };
  
        const newSuccessAlert = () => {
          return document
            .createRange()
              .createContextualFragment(`<vl-alert type="success" icon="check" title="Gelukt!" closable>
                                          <p>Wij hebben uw melding goed ontvangen en nemen deze spoedig in behandeling.</p>
                                         </vl-alert>`);
        }
      </script>
      <div id="top-left-toaster" is="vl-toaster" top-left fadeout></div>
      <button onclick="topLeftToaster().push(newErrorAlert());">Error!</button>
      <button onclick="topLeftToaster().push(newSuccessAlert());">Success!</button>
